    async function handleChat() {
        const input = document.getElementById('chat-input');
        const stream = document.getElementById('chat-stream');
        if (!input.value.trim()) return;

        stream.innerHTML += `<div class="chat-user p-5 ml-12 text-right mb-4 shadow-md font-medium text-sm transition-all animate-pulse">${msg}</div>`;

    }

            status: 'Applied', createdAt: serverTimestamp(), uid: currentUser.uid

    function loadLeads() {
        onSnapshot(query(getLeadsRef()), (snap) => {
            const grid = document.getElementById('leads-grid');
            if (!grid) return;
            grid.innerHTML = '';
            allLeads = [];
            snap.forEach(d => allLeads.push({ id: d.id, ...d.data() }));
            
            if (allLeads.length === 0) {
                grid.innerHTML = `<div class="col-span-full py-20 text-center text-slate-400 font-bold uppercase tracking-widest text-xs">No dossiers found.</div>`;
                return;
            }

            allLeads.sort((a, b) => (b.createdAt?.toMillis() || 0) - (a.createdAt?.toMillis() || 0));
            
            allLeads.forEach(lead => {
                const card = document.createElement('div');
                card.className = "bg-white p-8 rounded-[3rem] border border-slate-100 shadow-sm hover:shadow-2xl transition-all flex flex-col justify-between group";
                card.innerHTML = `
                    <div class="text-left">
                        <div class="flex justify-between items-start mb-4">
                            <h4 class="font-black text-2xl text-slate-900">${lead.name}</h4>
                            <span class="text-[9px] font-bold px-2 py-1 bg-blue-50 text-blue-600 rounded uppercase tracking-widest">${lead.type || 'Client'}</span>
                        </div>
                        <p class="text-sm text-slate-500 italic line-clamp-2 mb-6 text-left">"${lead.notes}"</p>
                    </div>
                    <button onclick="viewLead('${lead.id}')" class="w-full py-4 bg-slate-50 text-slate-900 rounded-2xl text-[10px] font-black uppercase tracking-widest group-hover:bg-blue-600 group-hover:text-white transition-all">Review AI Dossier ✨</button>
                `;
                grid.appendChild(card);
            });
        });
    }

    window.viewLead = (id) => {
        const lead = allLeads.find(l => l.id === id);
        openModal(`${lead.name}: Intelligence Briefing`, `<div class="text-slate-700 leading-relaxed font-medium text-left">${lead.ai_summary || 'Analysis pending...'}</div>`);
    };
